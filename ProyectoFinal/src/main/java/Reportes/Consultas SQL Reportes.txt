-->Tipo Actividad
SELECT COUNT(R.ID_ACT_REG), T.NOM_TIP
FROM REGISTROS R, TIPOS_DE_ACTIVIDAD T, ACTIVIDADES A
WHERE T.ID_TIP=A.TIP_ACT_PER
AND A.ID_ACT=R.ID_ACT_REG
GROUP BY T.NOM_TIP;

-->personal TI
SELECT COUNT(R.ID_USU_REG), U.NOM_USU
FROM REGISTROS R, USUARIOS U
WHERE U.USU=R.ID_USU_REG
GROUP BY U.NOM_USU;

-->estados

-->filtrado fechas
SELECT R.ID_REG, R. EST_REG, R.FEC_REG, R.HOR_REG, A.NOM_ACT, U.NOM_USU, U.APE_USU, E.NOM_EMP, E.APE_EMP, R.OBS_REG
FROM REGISTROS R, ACTIVIDADES A, USUARIOS U, EMPLEADOS E
WHERE A.ID_ACT=R.ID_ACT_REG
AND U.USU=R.ID_USU_REG
AND E.ID_EMP=R.ID_EMP_REG
AND FEC_REG BETWEEN $P{INI} AND $P{FIN};



ORDER BY FEC_REG DESC;
