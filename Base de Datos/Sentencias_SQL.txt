CREATE TABLE USUARIOS (
    ID_USU INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    USU VARCHAR(15) NOT NULL,
    PAS VARCHAR(15) NOT NULL,
    NOM_USU VARCHAR(15) NOT NULL,
    APE_USU VARCHAR(15) NOT NULL,
    COR_USU VARCHAR(25) NOT NULL
);

CREATE TABLE TIPOS_DE_ACTIVIDAD(
    ID_TIP INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    NOM_TIP VARCHAR(25) NOT NULL,
    DES_TIP VARCHAR(25) NOT NULL
);

CREATE TABLE ACTIVIDADES(
    ID_ACT INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    NOM_ACT VARCHAR(25) NOT NULL,
    DES_ACT VARCHAR(25) NOT NULL,
    TIP_ACT_PER INT NOT NULL REFERENCES TIPOS_DE_ACTIVIDAD(ID_TIP)
);

CREATE TABLE DEPARTAMENTOS(
    ID_DEP INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    NOM_DEP VARCHAR(25) NOT NULL
    DES_DEP VARCHAR(25) NOT NULL
)

CREATE TABLE EMPLEADOS(
    ID_EMP INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    NOM_EMP VARCHAR(25) NOT NULL,
    APE_EMP VARCHAR(25) NOT NULL,
    ID_DEP_PER INT NOT NULL REFERENCES DEPARTAMENTOS(ID_DEP)
);

CREATE TABLE REGISTROS(
    ID_REG INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    EST_REG VARCHAR(15) NOT NULL,
    FEC_REG DATE NOT NULL,
    HOR_REG VARCHAR(10) NOT NULL,
    ID_ACT_REG INT NOT NULL REFERENCES ACTIVIDADES(ID_ACT),
    ID_USU_REG INT NOT NULL REFERENCES USUARIOS(ID_USU),
    ID_EMP_REG INT NOT NULL REFERENCES EMPLEADOS(ID_EMP)
);

CREATE TABLE OBSERVACIONES(
    ID_REG_PER INT NOT NULL REFERENCES REGISTROS(ID_REG),
    DES_OBS VARCHAR(25) NOT NULL
);
